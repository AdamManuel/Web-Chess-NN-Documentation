import { Component } from '@angular/core';

@Component({

  templateUrl: './app.component.html',

      new Peice(1, 2),
      new Peice(1, 2),

    ];
    const row8 = [

    this.board = [

   * Gets the 90% of the height of the window
   */
  getHeight() {
    return ((window.innerHeight * 9) / 10 - 40) / 8 + 'px';
  }

  /**
   * Handles the logic when a user clicks a tile on the board
   * @param coorY The y coordinate (N in the game normaly)
   * @param coorX The x coordinate (Y in the game normaly)
   */
  onClick(coorX: number, coorY: number) {
    console.log(this.board[coorX][coorY].getName());
    console.log(coorX + ' ' + coorY);
    console.log(this.board[coorX][coorY].getMoves(coorX, coorY));
    this.possibleMoves = this.board[coorX][coorY].getMoves(coorX, coorY);
    console.log(this.possibleMoves);
  }

  /**
   * Gets the hugh of the tile for the players UI
   * @param coorY The y coordinate (N in the game normaly)
   * @param coorX The x coordinate (Y in the game normaly)
   */
  getHugh(coorX: number, coorY: number) {
    if (this.possibleMoves) {
      for (let index = 0; index < this.possibleMoves.length; index++) {
        const move = this.possibleMoves[index];
        if (move.coorX === coorX && move.coorY === coorY) {
          return 'rgba(0,255,0,.5)';
        }
      }
    } else {
      return 'rgba(0,0,0,0)';
    }
  }

  /**
   * The temporary function to handle when the heatmap buttons are clicked
   * @param button 
   */
  tempButton(button: number) {
    console.log('button ' + button + ' was pressed');
  }
}

/**
 * Every tile is considered a peice in this code. This is where the logic goes for each move
 */
export class Peice {
  /**
   * 
   * @param type type 0 = nothing | type 1 = pawn | type 2 = rook | type 3 = knight | type 4 = bishop | type 5 = queen | type 6 = king
   * @param team team 1 = white | team 2 = black | team 0 = None
   */
  constructor(public type?: number, public team?: number) {
    if (!type) {
      type = 0;
    }
    if (!team) {
      team = 0;
    }
  }

  /**
   * Gets the name of the peice on the board
   */
  getName() {
    switch (this.type) {
      case 0:
        return '';
      case 1:
        return 'pawn';
      case 2:
        return 'rook';
      case 3:
        return 'knight';
      case 4:
        return 'bishop';
      case 5:
        return 'queen';
      case 6:
        return 'king';
      default:
        return 'empty';
    }
  }

  /**
   * This returns the team associated with the 'peice'
   */
  getTeam() {
    if (this.team === 1) {
      return 'white';
    } else if (this.team === 2) {
      return 'black';
    } else {
      return 'none';
    }
  }

  /**
   * this gets the possible moves of the associated peice ignoring the other teams peices
   * @param coorX the current x coordinate of the peice
   * @param coorY the current y coordinate of the peice
   */
  getMoves(coorX: number, coorY: number) {
    switch (this.type) {
      case 0:
        return [];
      case 1:
        return this.getPawnMoves(coorX, coorY, this.team);
      case 2:
        return this.getRookMoves(coorX, coorY);
      case 3:
        return this.getKnightMoves(coorX, coorY);
      case 4:
        return this.getBishopMoves(coorX, coorY);
      case 5:
        return this.getQueenMoves(coorX, coorY);
      case 6:
        return this.getKingMoves(coorX, coorY, this.team);
      default:
        return [];
    }
  }

  /**
   * gets the moves of the pawn peice
   * @param coorX the x coordinate of the peice
   * @param coorY the y coordinate of the peice
   * @param team the team of the peice
   */
  getPawnMoves(coorX: number, coorY: number, team: number) {
    if (team === 1) {
      return [{ coorX: coorX + 1, coorY: coorY }];
    } else {
      return [{ coorX: coorX - 1, coorY: coorY }];
    }
  }

  /**
   * gets the moves of the rook peice
   * @param coorX the x coordinate of the peice
   * @param coorY the y coordinate of the peice
   */
  getRookMoves(coorX: number, coorY: number) {}

  /**
   * gets the moves of the knight peice
   * @param coorX the x coordinate of the peice
   * @param coorY the y coordinate of the peice
   */
  getKnightMoves(coorX: number, coorY: number) {}

  /**
   * gets the moves of the bishop peice
   * @param coorX the x coordinate of the peice
   * @param coorY the y coordinate of the peice
   */
  getBishopMoves(coorX: number, coorY: number) {}

  /**
   * gets the moves of the queen peice
   * @param coorX the x coordinate of the peice
   * @param coorY the y coordinate of the peice
   */
  getQueenMoves(coorX: number, coorY: number) {}

  /**
   * gets the moves of the king peice
   * @param coorX the x coordinate of the peice
   * @param coorY the y coordinate of the peice
   * @param team the team of the peice
   */
  getKingMoves(coorX: number, coorY: number, team: number) {}
}
